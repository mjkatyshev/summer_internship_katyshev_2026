#include <iostream>
#include <vector>
#include <algorithm>
#define all(x) x.begin(), x.end()
using namespace std;
using ll = long long;

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);

    ll n;
    cin >> n;

    vector<ll>numbers(n);

    for (int i = 0; i < n; i++) {
        cin >> numbers[i];
    }

    sort(all(numbers));

    ll count = 0;

    for (ll k = 2; k < n; k++) {
        ll i = 0, j = k - 1;
        while (i < j) {
            if (numbers[i] + numbers[j] > numbers[k]) {
                count += j - i;
                j--;
            } else {
                i += 1;
            }
        }
    }

    cout << count << endl;

    return 0;
}
