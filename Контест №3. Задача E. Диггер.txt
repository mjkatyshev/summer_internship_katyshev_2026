#include <iostream>
#include <vector>
#define all(x) x.begin(), x.end()
using namespace std;
using ll = long long;

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);

    ll n, m;
    cin >> n >> m;

    vector<vector<ll>>map(n, vector<ll>(m, 0));
    for (ll i = 0; i < n; i++) {
        for (ll j = 0; j < m; j++) {
            ll x;
            cin >> x;
            map[i][j] = x;
        }
    }

    vector<vector<ll>>paths(n, vector<ll>(m, 0));
    paths[0][0] = map[0][0];

    for (ll i = 0; i < n; i++) {
        for (ll j = 0; j < m; j++) {
            if (i == 0 && j == 0) continue;
            ll best = 0;
            if (i > 0) best = max(best, paths[i - 1][j]);
            if (j > 0) best = max(best, paths[i][j - 1]);
            if (i > 0 && j > 0) best = max(best, paths[i - 1][j - 1]);
            paths[i][j] = best + map[i][j];
        }
    }

    cout << paths[n - 1][m - 1] << endl;

    return 0;
}
