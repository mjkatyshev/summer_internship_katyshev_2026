#include <iostream>
#include <vector>
#define all(x) x.begin(), x.end()
using namespace std;
using ll = long long;

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);

    ll n, m;
    cin >> n >> m;

    vector<vector<ll>>g(n + 1);
    vector <bool>visited(n + 1, false);
    for (ll i = 0; i < m; i++) {
        ll u, v;
        cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u);
    }

    auto dfs = [&](auto&& dfs, ll u)->void {
        visited[u] = true;
        for (ll v : g[u]) {
            if (!visited[v]) dfs(dfs, v);
        }
        };

    ll count = 0;
    for (ll v = 1; v <= n; v++) {
        if (!visited[v]) {
            count++;
            dfs(dfs, v);
        }
    }

    
    cout << count << endl;

    return 0;
}
